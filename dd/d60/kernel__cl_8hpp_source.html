207 inline auto compile_kernel(const char* name,
208  const std::vector<std::string>& sources,
209  const std::map<std::string, int>& options) {
210  auto base_opts = opencl_context.base_opts();
211  for (auto& it : options) {
212  if (base_opts[it.first] > it.second) {
213  base_opts[it.first] = it.second;
214  }
215  }
216  std::string kernel_opts = "";
217  for (auto&& comp_opts : base_opts) {
218  kernel_opts += std::string(" -D") + comp_opts.first + "="
219  + std::to_string(comp_opts.second);
220  }
221  cl::Program program(opencl_context.context(), sources);
222  try {
223  program.build({opencl_context.device()}, kernel_opts.c_str());
224 
225  return cl::Kernel(program, name);
226  } catch (const cl::Error& e) {
227  // in case of CL_BUILD_PROGRAM_FAILURE, print the build error
228  if (e.err() == -11) {
229  std::string buildlog = program.getBuildInfo<CL_PROGRAM_BUILD_LOG>(
230  opencl_context.device()[0]);
231  system_error("compile_kernel", name, e.err(), buildlog.c_str());
232  } else {
233  check_opencl_error(name, e);
234  }
235  }
236  return cl::Kernel(); // never reached because check_opencl_error throws
237 }
