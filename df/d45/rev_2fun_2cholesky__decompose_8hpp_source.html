47 template <typename T1, typename T2, typename T3>
48 inline auto unblocked_cholesky_lambda(T1& L_A, T2& L, T3& A) {
49  return [L_A, L, A]() mutable {
50  const size_t N = A.rows();
51  // Algorithm is in rowmajor so we make the adjoint copy rowmajor
52  Eigen::Matrix<double, -1, -1, Eigen::RowMajor> adjL(L.rows(), L.cols());
53  Eigen::MatrixXd adjA = Eigen::MatrixXd::Zero(L.rows(), L.cols());
54  adjL.template triangularView<Eigen::Lower>() = L.adj();
55  for (int i = N - 1; i >= 0; --i) {
56  for (int j = i; j >= 0; --j) {
57  if (i == j) {
58  adjA.coeffRef(i, j) = 0.5 * adjL.coeff(i, j) / L_A.coeff(i, j);
59  } else {
60  adjA.coeffRef(i, j) = adjL.coeff(i, j) / L_A.coeff(j, j);
61  adjL.coeffRef(j, j)
62  -= adjL.coeff(i, j) * L_A.coeff(i, j) / L_A.coeff(j, j);
63  }
64  for (int k = j - 1; k >= 0; --k) {
65  adjL.coeffRef(i, k) -= adjA.coeff(i, j) * L_A.coeff(j, k);
66  adjL.coeffRef(j, k) -= adjA.coeff(i, j) * L_A.coeff(i, k);
67  }
68  }
69  }
70  A.adj() += adjA;
71  };
72 }

80 template <typename T1, typename T2, typename T3>
81 inline auto cholesky_lambda(T1& L_A, T2& L, T3& A) {
82  return [L_A, L, A]() mutable {
83  using Block_ = Eigen::Block<Eigen::MatrixXd>;
84  using Eigen::Lower;
85  using Eigen::StrictlyUpper;
86  using Eigen::Upper;
87  Eigen::MatrixXd L_adj = Eigen::MatrixXd::Zero(L.rows(), L.cols());
88  L_adj.template triangularView<Eigen::Lower>() = L.adj();
89  const int M_ = L_A.rows();
90  int block_size_ = std::max(M_ / 8, 8);
91  block_size_ = std::min(block_size_, 128);
92  for (int k = M_; k > 0; k -= block_size_) {
93  int j = std::max(0, k - block_size_);
94  auto R = L_A.block(j, 0, k - j, j);
95  auto D = L_A.block(j, j, k - j, k - j).eval();
96  auto B = L_A.block(k, 0, M_ - k, j);
97  auto C = L_A.block(k, j, M_ - k, k - j);
98  auto R_adj = L_adj.block(j, 0, k - j, j);
99  auto D_adj = L_adj.block(j, j, k - j, k - j);
100  auto B_adj = L_adj.block(k, 0, M_ - k, j);
101  auto C_adj = L_adj.block(k, j, M_ - k, k - j);
102  D.transposeInPlace();
103  if (C_adj.size() > 0) {
104  C_adj = D.template triangularView<Upper>()
105  .solve(C_adj.transpose())
106  .transpose();
107  B_adj.noalias() -= C_adj * R;
108  D_adj.noalias() -= C_adj.transpose() * C;
109  }
110  D_adj = (D * D_adj.template triangularView<Lower>()).eval();
111  D_adj.template triangularView<StrictlyUpper>()
112  = D_adj.adjoint().template triangularView<StrictlyUpper>();
113  D.template triangularView<Upper>().solveInPlace(D_adj);
114  D.template triangularView<Upper>().solveInPlace(D_adj.transpose());
115  R_adj.noalias() -= C_adj.transpose() * B;
116  R_adj.noalias() -= D_adj.template selfadjointView<Lower>() * R;
117  D_adj.diagonal() *= 0.5;
118  }
119  A.adj().template triangularView<Eigen::Lower>() += L_adj;
120  };
121 }
