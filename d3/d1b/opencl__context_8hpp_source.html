74  opencl_context_base(int platform_id = OPENCL_PLATFORM_ID,
75  int device_id = OPENCL_DEVICE_ID) {
76  try {
77  // platform
78  cl::Platform::get(&platforms_);
79  if (platform_id >= platforms_.size()) {
80  system_error("OpenCL Initialization", "[Platform]", -1,
81  "CL_INVALID_PLATFORM");
82  }
83  platform_.push_back(platforms_[platform_id]);
84  platform_name_ = platform_[0].getInfo<CL_PLATFORM_NAME>();
85  platform_[0].getDevices(DEVICE_FILTER, &devices_);
86  if (devices_.size() == 0) {
87  system_error("OpenCL Initialization", "[Device]", -1,
88  "CL_DEVICE_NOT_FOUND");
89  }
90  if (device_id >= devices_.size()) {
91  system_error("OpenCL Initialization", "[Device]", -1,
92  "CL_INVALID_DEVICE");
93  }
94  device_.push_back(devices_[device_id]);
95  // context and queue
96  cl_command_queue_properties device_properties;
97  device_[0].getInfo<cl_command_queue_properties>(
98  CL_DEVICE_QUEUE_PROPERTIES, &device_properties);
99  device_[0].getInfo<size_t>(CL_DEVICE_MAX_WORK_GROUP_SIZE,
100  &max_thread_block_size_);
101  std::vector<size_t> max_wg_sizes
102  = device_[0].getInfo<CL_DEVICE_MAX_WORK_ITEM_SIZES>();
103  if (max_wg_sizes.size() < 3) {
104  system_error("OpenCL Initialization", "[Device]", -1,
105  "The device does not support 3D work groups!");
106  }
107 
108  context_ = cl::Context(device_[0]);
109  if (device_properties & CL_QUEUE_OUT_OF_ORDER_EXEC_MODE_ENABLE) {
110  command_queue_
111  = cl::CommandQueue(context_, device_[0],
112  CL_QUEUE_OUT_OF_ORDER_EXEC_MODE_ENABLE, nullptr);
113  in_order_ = CL_FALSE;
114  } else {
115  command_queue_ = cl::CommandQueue(context_, device_[0], 0, nullptr);
116  in_order_ = CL_TRUE;
117  }
118  int max_square_block_size
119  = std::min({max_wg_sizes[0], max_wg_sizes[1],
120  static_cast<size_t>(std::sqrt(max_thread_block_size_))});
121 
122  // Does a compile time check of the maximum allowed
123  // dimension of a square thread block size
124  // WG size of (32,32) works on all recent GPUs but would fail on some
125  // older integrated GPUs or CPUs
126  if (max_square_block_size < base_opts_["THREAD_BLOCK_SIZE"]) {
127  base_opts_["THREAD_BLOCK_SIZE"] = max_square_block_size;
128  base_opts_["WORK_PER_THREAD"] = 1;
129  }
130  if (std::min(max_thread_block_size_, max_wg_sizes[0])
131  < base_opts_["LOCAL_SIZE_"]) {
132  // must be a power of base_opts_["REDUCTION_STEP_SIZE"]
133  const int p = std::log(max_thread_block_size_)
134  / std::log(base_opts_["REDUCTION_STEP_SIZE"]);
135  base_opts_["LOCAL_SIZE_"]
136  = std::pow(base_opts_["REDUCTION_STEP_SIZE"], p);
137  }
138  // Thread block size for the Cholesky
139  // TODO(Steve): This should be tuned in a higher part of the stan language
140  if (max_thread_block_size_ >= 256) {
141  tuning_opts_.cholesky_min_L11_size = 256;
142  } else {
143  tuning_opts_.cholesky_min_L11_size = max_thread_block_size_;
144  }
145  } catch (const cl::Error& e) {
146  check_opencl_error("opencl_context", e);
147  }
148  }
