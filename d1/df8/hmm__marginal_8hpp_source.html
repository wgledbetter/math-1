20 inline auto hmm_marginal_val(
21  const Eigen::Matrix<T_omega, Eigen::Dynamic, Eigen::Dynamic>& omegas,
22  const T_Gamma& Gamma_val, const T_rho& rho_val,
23  Eigen::Matrix<T_alpha, Eigen::Dynamic, Eigen::Dynamic>& alphas,
24  Eigen::Matrix<T_alpha, Eigen::Dynamic, 1>& alpha_log_norms,
25  T_alpha& norm_norm) {
26  const int n_states = omegas.rows();
27  const int n_transitions = omegas.cols() - 1;
28  alphas.col(0) = omegas.col(0).cwiseProduct(rho_val);
29 
30  const auto norm = alphas.col(0).maxCoeff();
31  alphas.col(0) /= norm;
32  alpha_log_norms(0) = log(norm);
33 
34  auto Gamma_val_transpose = Gamma_val.transpose().eval();
35  for (int n = 1; n <= n_transitions; ++n) {
36  alphas.col(n)
37  = omegas.col(n).cwiseProduct(Gamma_val_transpose * alphas.col(n - 1));
38  const auto col_norm = alphas.col(n).maxCoeff();
39  alphas.col(n) /= col_norm;
40  alpha_log_norms(n) = log(col_norm) + alpha_log_norms(n - 1);
41  }
42  norm_norm = alpha_log_norms(n_transitions);
43  return log(alphas.col(n_transitions).sum()) + norm_norm;
