148 struct FixedPointADJac {
157  template <typename F>
158  inline Eigen::Matrix<stan::math::var, -1, 1> operator()(
159  const Eigen::VectorXd& x, const Eigen::Matrix<stan::math::var, -1, 1>& y,
160  KinsolFixedPointEnv<F>& env) {
161  using stan::math::precomputed_gradients;
162  using stan::math::to_array_1d;
163  using stan::math::var;
164 
165  auto g = [&env](const Eigen::Matrix<var, -1, 1>& par_) {
166  Eigen::Matrix<var, -1, 1> x_(par_.head(env.N_));
167  Eigen::Matrix<var, -1, 1> y_(par_.tail(env.M_));
168  return env.f_(x_, y_, env.x_r_, env.x_i_, env.msgs_);
169  };
170 
171  Eigen::VectorXd theta(x.size() + y.size());
172  for (int i = 0; i < env.N_; ++i) {
173  theta(i) = x(i);
174  }
175  for (int i = 0; i < env.M_; ++i) {
176  theta(i + env.N_) = env.y_(i);
177  }
178  Eigen::Matrix<double, -1, 1> fx;
179  Eigen::Matrix<double, -1, -1> J_theta;
180  stan::math::jacobian(g, theta, fx, J_theta);
181  Eigen::MatrixXd A(J_theta.block(0, 0, env.N_, env.N_));
182  Eigen::MatrixXd b(J_theta.block(0, env.N_, env.N_, env.M_));
183  A = Eigen::MatrixXd::Identity(env.N_, env.N_) - A;
184  Eigen::MatrixXd Jxy = A.colPivHouseholderQr().solve(b);
185  std::vector<double> gradients(env.M_);
186  Eigen::Matrix<var, -1, 1> x_sol(env.N_);
187  std::vector<stan::math::var> yv(to_array_1d(y));
188  for (int i = 0; i < env.N_; ++i) {
189  gradients = to_array_1d(Eigen::VectorXd(Jxy.row(i)));
190  x_sol[i] = precomputed_gradients(x(i), yv, gradients);
191  }
192  return x_sol;
193  }
