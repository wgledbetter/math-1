98  inline void operator()(const tbb::blocked_range<size_t>& r) {
99  if (r.empty()) {
100  return;
101  }
102 
103  if (args_adjoints_.size() == 0) {
104  args_adjoints_ = Eigen::VectorXd::Zero(num_vars_shared_terms_);
105  }
106 
107  // Obtain reference to a local copy of all shared arguments that do
108  // not point
109  // back to main autodiff stack
110 
111  if (!local_args_tuple_scope_.args_tuple_holder_) {
112  // shared arguments need to be copied to reducer-specific
113  // scope. In this case no need for zeroing adjoints, since the
114  // fresh copy has all adjoints set to zero.
115  local_args_tuple_scope_.stack_.execute([&]() {
116  apply(
117  [&](auto&&... args) {
118  local_args_tuple_scope_.args_tuple_holder_ = std::make_unique<
119  typename scoped_args_tuple::args_tuple_t>(
120  deep_copy_vars(args)...);
121  },
122  args_tuple_);
123  });
124  } else {
125  // set adjoints of shared arguments to zero
126  local_args_tuple_scope_.stack_.execute([] { set_zero_all_adjoints(); });
127  }

132  const nested_rev_autodiff begin_nest;

134  // Create nested autodiff copies of sliced argument that do not point
135  // back to main autodiff stack
136  std::decay_t<Vec> local_sub_slice;
137  local_sub_slice.reserve(r.size());
138  for (size_t i = r.begin(); i < r.end(); ++i) {
139  local_sub_slice.emplace_back(deep_copy_vars(vmapped_[i]));
140  }

145  return ReduceFunction()(local_sub_slice, r.begin(), r.end() - 1,

157  accumulate_adjoints(sliced_partials_ + r.begin() * num_vars_per_term_,
158  std::move(local_sub_slice));
