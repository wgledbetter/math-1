44  y_(ChainableStack::instance_->memalloc_.alloc_array<vari*>(y.size())),

47  theta_(
48  ChainableStack::instance_->memalloc_.alloc_array<vari*>(x_size_)),
49  Jx_y_(ChainableStack::instance_->memalloc_.alloc_array<double>(
50  x_size_ * y_size_)) {
51  using Eigen::Map;
52  using Eigen::MatrixXd;
53  for (int i = 0; i < y.size(); ++i) {
54  y_[i] = y(i).vi_;
55  }
56 
57  theta_[0] = this;
58  for (int i = 1; i < x.size(); ++i) {
59  theta_[i] = new vari(theta_dbl(i), false);
60  }
61 
62  // Compute the Jacobian and store in array, using the
63  // implicit function theorem, i.e. Jx_y = Jf_y / Jf_x
64  using f_y = hybrj_functor_solver<Fs, F, double, double>;
65  Map<MatrixXd>(&Jx_y_[0], x_size_, y_size_)
66  = -mdivide_left(fx.get_jacobian(theta_dbl),
67  f_y(fs, f, theta_dbl, value_of(y), dat, dat_int, msgs)
68  .get_jacobian(value_of(y)));
69  }

71  void chain() {
72  for (int j = 0; j < y_size_; j++) {
73  for (int i = 0; i < x_size_; i++) {
74  y_[j]->adj_ += theta_[i]->adj_ * Jx_y_[j * x_size_ + i];
75  }
76  }
77  }

126 Eigen::VectorXd algebra_solver_powell(
127  const F& f, const T& x, const Eigen::VectorXd& y,
128  const std::vector<double>& dat, const std::vector<int>& dat_int,
129  std::ostream* msgs = nullptr, double relative_tolerance = 1e-10,
130  double function_tolerance = 1e-6,
131  long int max_num_steps = 1e+3) { // NOLINT(runtime/int)
132  const auto& x_eval = x.eval();
133  const auto& x_val = (value_of(x_eval)).eval();
134  algebra_solver_check(x_val, y, dat, dat_int, function_tolerance,
135  max_num_steps);
136  check_nonnegative("alegbra_solver", "relative_tolerance", relative_tolerance);
137  // if (relative_tolerance < 0)
138  // invalid_argument("algebra_solver", "relative_tolerance,",
139  // function_tolerance, "",
140  // ", must be greater than or equal to 0");
141 
142  // Create functor for algebraic system
143  using Fs = system_functor<F, double, double, true>;
144  using Fx = hybrj_functor_solver<Fs, F, double, double>;
145  Fx fx(Fs(), f, x_val, y, dat, dat_int, msgs);
146  Eigen::HybridNonLinearSolver<Fx> solver(fx);
147 
148  // Check dimension unknowns equals dimension of system output
149  check_matching_sizes("algebra_solver", "the algebraic system's output",
150  fx.get_value(x_val), "the vector of unknowns, x,", x);
151 
152  // Compute theta_dbl
153  Eigen::VectorXd theta_dbl = x_val;
154  solver.parameters.xtol = relative_tolerance;
155  solver.parameters.maxfev = max_num_steps;
156  solver.solve(theta_dbl);
157 
158  // Check if the max number of steps has been exceeded
159  if (solver.nfev >= max_num_steps) {
160  throw_domain_error("algebra_solver", "maximum number of iterations",
161  max_num_steps, "(", ") was exceeded in the solve.");
162  }
163 
164  // Check solution is a root
165  double system_norm = fx.get_value(theta_dbl).stableNorm();
166  if (system_norm > function_tolerance) {
167  std::ostringstream message;
168  message << "the norm of the algebraic function is " << system_norm
169  << " but should be lower than the function "
170  << "tolerance:";
171  throw_domain_error("algebra_solver", message.str().c_str(),
172  function_tolerance, "",
173  ". Consider decreasing the relative tolerance and "
174  "increasing max_num_steps.");
175  }
176 
177  return theta_dbl;
178 }
