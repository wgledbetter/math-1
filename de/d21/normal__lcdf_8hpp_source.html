69  const T_partials_return x2 = square(scaled_diff);

71  // Rigorous numerical approximations are applied here to deal with values
72  // of |scaled_diff|>>0. This is needed to deal with rare base-rate
73  // logistic regression problems where it is useful to use an alternative
74  // link function instead.
75  //
76  // use erfc() instead of erf() in order to retain precision
77  // since for x>0 erfc()->0
78  if (scaled_diff > 0.0) {
79  // CDF(x) = 1/2 + 1/2erf(x) = 1 - 1/2erfc(x)
80  cdf_log += log1p(-0.5 * erfc(scaled_diff));
81  if (!is_not_nan(cdf_log)) {

83  }
84  } else if (scaled_diff > -20.0) {

86  cdf_log += log(erfc(-scaled_diff)) + LOG_HALF;
87  } else if (10.0 * log(fabs(scaled_diff))
88  < log(std::numeric_limits<T_partials_return>::max())) {
89  // entering territory where erfc(-x)~0
90  // need to use direct numerical approximation of cdf_log instead
91  // the following based on W. J. Cody, Math. Comp. 23(107):631-638 (1969)
92  // CDF(x) = 1/2erfc(-x)
93  const T_partials_return x4 = pow(scaled_diff, 4);
94  const T_partials_return x6 = pow(scaled_diff, 6);
95  const T_partials_return x8 = pow(scaled_diff, 8);
96  const T_partials_return x10 = pow(scaled_diff, 10);

98  = 0.000658749161529837803157 + 0.0160837851487422766278 / x2
99  + 0.125781726111229246204 / x4 + 0.360344899949804439429 / x6
100  + 0.305326634961232344035 / x8 + 0.0163153871373020978498 / x10;

102  = -0.00233520497626869185443 - 0.0605183413124413191178 / x2
103  - 0.527905102951428412248 / x4 - 1.87295284992346047209 / x6
104  - 2.56852019228982242072 / x8 - 1.0 / x10;
105  cdf_log += LOG_HALF + log(INV_SQRT_PI + (temp_p / temp_q) / x2)
106  - log(-scaled_diff) - x2;
107  } else {

109  cdf_log = stan::math::negative_infinity();

111

114  // based on analytic form given by:
115  // dln(CDF)/dx = exp(-x^2)/(sqrt(pi)*(1/2+erf(x)/2)
116  T_partials_return dncdf_log = 0.0;

118  T_partials_return t2 = 0.0;

122  // (due to instability / inaccuracy in the various approximations)
123  if (scaled_diff > 2.9) {
124  // approximation derived from Abramowitz and Stegun (1964) 7.1.26
125  t = 1.0 / (1.0 + 0.3275911 * scaled_diff);

132  + 1.453152027 * t2 * t - 1.061405429 * t4));
133  } else if (scaled_diff > 2.5) {
134  // in the trouble area where all of the standard numerical
135  // approximations are unstable - bridge the gap using Taylor
136  // expansions of the analytic function
137  // use Taylor expansion centred around x=2.7
138  t = scaled_diff - 2.7;

140  t4 = pow(t, 4);
